JavaScript事件_Event

Event对象

Event 对象代表事件的状态，比如事件在其中发生的元素、键盘按键的状态、鼠标的位置、鼠标按钮的状态。

Event属性

常见的有onclick 点击某个对象时调用事件

//假设你要点击按钮使字体变色
var input=document.getElementById("input");
var box=document.getElementById("box");
input.onclick=function(){
	box.style.color=red;
}

ondblclick 双击某个对象时调用事件

onfocus 元素获得焦点

//在获取焦点时才能动作
var input=document.getElementById("input");
input.onfocus=function (){
	...里面写你想执行的动作
}

onkeydown 某个键盘按键被按下

//你按下某个键盘显示它的键值
function (e){
var m=event.keyCode;
e.onkeydown=m;
}

onmousemove 鼠标移动

//鼠标在窗口移动获取它的XY值
window.onmousemove=function move(e){
sp1.innerText=e.clientX;
sp2.innerText=e.clientY;
}

addEventListener

//也可以用这个来运用事件，但有on的事件要去掉on
var box=document.getElementById("box");
box.addEventListener("click",dianji,false);
//括号里的第一个元素表示事件，但要去掉on，第二个表示执行的函数，第三个表示采用事件流还是冒泡流，false表示冒泡流，true表示事件流，不写默认是false
dianji(){
	...执行的动作
}

事件流与冒泡流

好比上级领导发布命令，向下级层层传递，得到寄过以后再层层向上反馈

事件流只会在最后响应，就是只有目标会作出动作，而冒泡流则是它所有父级都会作出动作

	  //这是一个点击事件，点击以后会倒计时，如下图
var btn=document.getElementById("btn");
	btn.addEventListener("click",move,false);
	var t=5;
	function move(){				
		btn.innerHTML=t+"秒后重新获取";
		btn.disabled=true;
		siv1=window.setInterval(time,1000);
	}
	function time(){
		t--;
		btn.innerHTML=t+"秒后重新获取";
		if(t<=0){
		btn.innerHTML="获取手机验证码";
		btn.disabled=false;
		t=5;
		clearInterval(siv1);
	}
	
}
